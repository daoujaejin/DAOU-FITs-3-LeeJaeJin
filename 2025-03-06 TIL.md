	// 최대값 찾기
	int max = (a > b) ? ((a > c) ? ((a > d) ? a : d) : ((c > d) ? c : d))
		: ((b > c) ? ((b > d) ? b : d) : ((c > d) ? c : d));
	// 최소값 찾기
	int min = (a < b) ? ((a < c) ? ((a < d) ? a : d) : ((c < d) ? c : d))
		: ((b < c) ? ((b < d) ? b : d) : ((c < d) ? c : d));
	printf("최대값: %d, 최소값: %d\n", max, min);
}

int last() {

	printf("정수 입력: ");
	scanf_s("%d", &num);

	if (num == 0) {
		printf("유효비트: 1\n");
		return 0;
	}
	int bits = 0;
	while (num > 0) {
		bits++;
		num >>= 1;
	}
	printf("유효비트: %d\n", bits);
