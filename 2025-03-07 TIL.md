void p4_p1() {
    printf("\n[4페이지 1번]\n");
    char male, real_man, marry;
    int child, score = 0;

    printf("성별을 말해주세요(M / F): ");
    scanf_s("%c", &male, 1);
    getchar();
    switch(male) {
        case 'M':
            printf("군 제대 여부 (O / X): ");
            scanf_s("%c", &real_man, 1);
            if (real_man == 'O') {
                score++;
                printf("기혼 여부 (O / X): ");
                scanf_s("%c", &marry, 1);
                getchar();
                if (marry == 'O')
                    score++;
            }
            printf("점수: %d\n", score);
            break;
        case 'F':
            printf("기혼 여부 (O / X): ");
            scanf_s("%c", &marry, 1);
            getchar();
            if (marry == 'O') {
                score++;
                printf("자녀 수: ");
                scanf_s("%d", &child);
                score += child;
            }
            printf("점수: %d\n", score);
    }
}
void p6_p10() {
    printf("\n[6페이지 10번]\n");
    char small = 0;
    while (small < 97 || small > 122) {
        printf("알파벳 소문자를 입력하시요\n");
        // scanf_s("%c", &small, 1);
        small = _getch();
        if (small == 0x1b) {
            break;
        }
        if (small >= 97 && small <= 122) {
            printf("변환된 대문자는 %c입니다.\n", small - 32);
            break;
        } else {
            printf("\n");
        }
    }
}

    if (x < y) {
        int temp;
        temp = y;
        y = x;
        x = temp;
    }
    r = x;
    while (y != 0) {
        r = x % y;
        x = y;
        y = r;
    }
    printf("최대공약수는 %d입니다.\n", x);
}

void func_4(int year) {
    if (leapYear(year))
        printf("총 일은 366일입니다.\n");
    else
        printf("총 일은 365일입니다.\n");
}

bool leapYear(int year) {
    if ((year % 4 == 0 && year % 100 != 0) || year % 400 == 0) {
        return 1; // 윤년이면 1 반환
    } else {
        return 0; // 윤년이 아니면 0 반환
    }
}

int rec_6(int n) {
    if (n == 1)
        return 1;
    else
        return rec_6(n - 1) + n;
}

int rec_7(int n) {
    if (n == 2)
        return 2;
    else
        return rec_7(n - 1) * n;
}

int rec_8(int n) {
    if (n == 1) {
        return 0;
    } else if (n == 2) {
        return 1;
    } else {
        int now = rec_8(n - 1) + rec_8(n - 2);
        return now;
    }
}

int rec_9(float n, int fold) {
    // foldPaper
    printf("%d번 접었을 때의 넓이는 %f\n", fold, n);
    if (n < 1) {
        return 0;
    } else {
        return rec_9(n / 2, fold + 1) + 1;
    }
}
