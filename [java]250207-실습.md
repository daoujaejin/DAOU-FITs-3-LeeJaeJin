package javaexam2.plane;

public abstract class Plane {
    private String  planeName;
    private int     fuelSize;

    public Plane() {

    }
    public Plane(String planeName, int fuelSize) {
        this.planeName = planeName;
        this.fuelSize = fuelSize;
    }

    public void setFuelSize(int fuelSize) {
        this.fuelSize = fuelSize;
    }
    public void setPlaneName(String planeName) {
        this.planeName = planeName;
    }
    public int getFuelSize() {
        return fuelSize;
    }
    public String getPlaneName() {
        return planeName;
    }

    public void refuel(int fuel) {
        this.fuelSize += fuel;
    }

    public abstract void flight(int distance);
}

package javaexam2.company;

public abstract class Employee {
    private String  name;
    private int     number;
    private String  department;
    private int     salary;

    public Employee() {

    }
    public Employee(String name, int number, String department, int salary) {
        this.name = name;
        this.number = number;
        this.department = department;
        this.salary = salary;
    }

    public void setName(String name) {
        this.name = name;
    }
    public void setNumber(int number) {
        this.number = number;
    }
    public void setDepartment(String department) {
        this.department = department;
    }

    public void setSalary(int salary) {
        this.salary = salary;
    }
    public String getName() {
        return name;
    }
    public int getNumber() {
        return number;
    }
    public String getDepartment() {
        return department;
    }
    public int getSalary() {
        return salary;
    }

    abstract public double tax();

    @Override
    public String toString() {
        return name + " " + department + " " + salary;
    }
}

package javaexam2.company;

import java.util.HashMap;
import java.util.Map;

public class Company {
    public static void main(String[] args) {
        Map<Integer, Employee> map = new HashMap<Integer, Employee>();

        // 데이터를 기반으로 객체를 생성하여 HashMap에 넣는다.
        map.put(1, new Secretary("Hilery ", 1, "secretary", 800));
        map.put(2, new Sales("Clinten", 2, "sales    ", 1200));

        // 모든 객체의 정보를 출력한다. (for문)
        System.out.println("name       department   salary");
        System.out.println("------------------------------");
        // for (int i = 1; i <= 2; i++) {
        for (Integer i : map.keySet()) {
            System.out.println(map.get(i).getName() + "    " + map.get(i).getDepartment() + "    " +
                    map.get(i).getSalary());
        }
        System.out.println();

        // 모든 객체에 인센티브 100씩 지급하고 다시 Hashmap에 넣는다.
        System.out.println("인센티브 100 지급");
        System.out.println();
        // ((Secretary)map.get(1)).incentive(100);
        // ((Sales)map.get(2)).incentive(100);
        for (Integer i : map.keySet()) {
            if (map.get(i) instanceof Secretary) {
                Secretary s = (Secretary)(map.get(i));
                s.incentive(100);
                map.put(s.getNumber(), (Employee)s);
            } else {
                Sales s = (Sales)(map.get(i));
                s.incentive(100);
                map.put(s.getNumber(), (Employee)s);
            }
        }

        // 모든 정보와 세금을 출력한다.
        System.out.println("name       department   salary   tax");
        System.out.println("--------------------------------------");
        for (Integer i : map.keySet()) {
            System.out.println(map.get(i).getName() + "    " + map.get(i).getDepartment() + "    " +
                    String.format("%-4s", map.get(i).getSalary()) + "     " + String.format("%-4s",map.get(i).tax()));
        }
    }
}
